In this section we will recall infinitesimal types and tangent spaces as developed by Cherubini et al. \cite{diffgeo-article}.
We will also develop infinitesimally linear types in SAG, following Myers \cite{david-orbifolds}.

\begin{definition}\label{def:hood-res}
  Let $(X,x)$ be a pointed type.
  We define \notion{the $n$-th order neighbourhood around $x$} as the subtype $\hood{X,x}{n} \defeq \setbuild{x' : X}{\exists I \substack{\subseteq \\ \mathrm{fg.ideal}} R, I^{n+1} = 0, I = 0 \rightarrow x' = x}$.
\end{definition}

Observe that there is a natural inclusion $i : \hood{X,x}{n} \hookrightarrow X$.

Let $f : (X,x) \rightarrow_\ast (Y,y)$ be a pointed map.
Restricting $f$ to the $n$-th order neighbourhood of $X$ around $x$ yields a map $f\mid_{\hood{X,x}{n}} : \hood{X,x}{n} \rightarrow_\ast \hood{Y,y}{n}$.
This gives a functor $\hood{\argmhole}{n} : \pType \rightarrow \pType$ which preserves embeddings.

\begin{definition}\label{def:hood}
  \rednote{Is there a better definition for this that does not depend on $x$?}
  A type $X$ is an \notion{$n$-th order neighbourhood around $x : X$} if the inclusion $i : \hood{X,x}{n} \hookrightarrow X$ is an equivalence of types.
\end{definition}

\begin{example}\label{ex:R[x]/(x^{n+1})}
  Consider the type $X \defeq \Spec(\quot{R[x_1, \dots, x_m]}{\ideal{x_1, \dots, x_m}^{n+1}})$.
  The ideal $\ideal{x_1, \dots, x_m}^{n+1}$ is generated by all monomials of total degree $n+1$.
  A point $f : X$ is an $R$-algebra homomorphism with codomain $R$, determined by the values $f(x_i) = r_i \in R$ for $i \in \set{1,\dots,m}$.
  The imposed relations require that for any monomial $p(x_1, \dots, x_m)$ of degree $n+1$, $f(p(x_1, \dots, x_m)) = p(r_1, \dots, r_m) = 0$.
  Equivalently, the ideal $I \defeq \ideal{r_1, \dots, r_m} \subseteq R$ satisfies $I^{n+1} = 0$.

  Now $X$ is an $n$-th order neighbourhood around $0$, i.e. the map $i : \hood{X,0}{n} \hookrightarrow X$ is an equivalence.
  To see this, consider $f : X$ given by $(r_1, \dots, r_m)$.
  If we let $I \defeq \ideal{r_1, \dots, r_m}$, then we have seen that $I^{n+1} = 0$.
  If now $I = 0$, it is clear that each $r_i=0$, so $f = 0$ as well.
\end{example}

In the preceding example, we will give a notation for the case $n=1$.
We define the \notion{$m$-dimensional infinitesimal disk}, or just the \notion{$m$-disk}, as the type $\D(m) \defeq \Spec(\quot{R[x_1,\dots, x_m]}{\ideal{x_1,\dots,x_m}^2})$.
The $m$-disk is a $1$-st order neighbourhood around $0$.
Thus, whenever we consider $\D (m)$ as a pointed type, we will always mean the pointed type $(\D (m), 0)$.

We will now introduce some notation.
Let $(X,x)$ and $(Y,y)$ be pointed types.
We define $(Y,y)^{(X,x)} \defeq (X,x) \rightarrow_\ast (Y,y)$.
If the pointing on $X$ is understood from context, then we will simply write $(Y,y)^X$.

\begin{definition}\label{def:tangent}
  Let $(X,x)$ be a pointed type.
  The tangent space of $X$ at $x$ is the type $\tangent{X}{x} \defeq (X,x)^{\D(1)}$.
\end{definition}

The tangent space construction defines a functor $\tangent{}{} : \pType \rightarrow \pType$.
If $f : (X,x) \rightarrow_\ast (Y,y)$ is a pointed map, the derivative of $f$ is the map $\der{f}{x} \defeq f_\ast : \tangent{X}{x} \rightarrow_\ast \tangent{Y}{y}$ induced by post-composition.
Taking tangent spaces is a local construction in the following sense.

\begin{lemma}\label{prp:hood-factorization}
  Let $(X,x)$ and $(Y,y)$ be pointed types.
  If $X$ is an $n$-th order neighbourhood around $x$, then every pointed map $f : (X,x) \rightarrow_\ast (Y,y)$ factors uniquely through $\hood{Y,y}{n}$.
\end{lemma}

\begin{proof}
  Let $x' : X$.
  Since $X$ is an $n$-th order neighbourhood around $x$, we merely have a finitely generated ideal $I \subseteq R$ such that $I^{n+1} = 0$ and if $I = 0$ then $x' = x$.
  It follows that $f(x') \in \hood{Y,y}{n}$, as if $I = 0$, then $f(x') = f(x) = y$ by applying $f$ to $x' = x$ and by $f$ being pointed.
\end{proof}

\begin{remark}\label{rem:tangent-factorization}
  Let $(X,x)$ be a pointed type.
  Specializing \Cref{prp:hood-factorization} to $\tangent{X}{x} \defeq (X,x)^{\D(1)}$, we observe that any tangent vector $\nu : \tangent{X}{x}$ has its image entirely in $\hood{X,x}{1}$.
  Thus, $\tangent{X}{x} = \tangent{\hood{X,x}{1}}{x}$.
\end{remark}

Let $M$ be an $R$-module.
The tangent space $\tangent{M}{0}$ admits an $R$-module structure, given by taking pointwise addition and pointwise scalar multiplication.
We call this the \notion{pointwise module structure} on $\tangent{M}{0}$.
Taking the derivative of any linear map $f : M \rightarrow N$ yields a linear map $\der{f}{0} : \tangent{M}{0} \rightarrow \tangent{N}{0}$.
We define a map $\can : M \rightarrow \tangent{M}{0}$ by $m \mapsto (r \mapsto r \cdot m)$, and we call this the \notion{canonical map}.
It is clear that the canonical map is $R$-linear with respect to the pointwise module structure.

\begin{proposition}\label{prp:can-nat}
  The family of maps $\mathrm{can} : M \rightarrow \mathrm{T}_0M$ is natural with respect to $R$-linear maps. \earlyqed
\end{proposition}

\begin{definition}\label{def:tangential}
  \rednote{this name should be debated: tangential vs. canonical?}
  An $R$-module $M$ is \notion{tangential} if $\can : M \rightarrow \tangent{M}{0}$ is an equivalence.
\end{definition}

\begin{example}\label{ex:R-lin-flat}
  The $R$-modules $R^n$ are tangential \cite{diffgeo-article}[Lemma 2.2.3].
\end{example}

\begin{proposition}\label{prp:exp-tangential}
  Let $M$ be a tangential $R$-module.
  Then for any type $U$, the $R$-module $M^U$ is tangential.
\end{proposition}

\begin{proof}
  This follows by symmetry of the cartesian product, i.e. $\tangent{(M^U)}{0} = \tangent{M}{0}^U$.
  The latter type is equivalent to $M^U$ by assumption.
\end{proof}

\begin{example}\label{ex:power-series-tangential}
    The modules $R[x] = R^R$ and $R[[x]] = R^\mathbb{N}$ are tangential.
\end{example}

Even if a type $X$ is not an $R$-module, the tangent space $\tangent{X}{x}$ may have an $R$-module structure.
This is the case whenever $X$ is ``infinitesimally linear'' at $x$.
We will now define this property.
For any $m,n : \N$ we define the following inclusion
\begin{align*}
  \inc_{m,n} : \D (m) \underset{\ast}{\coprod} \D (n) & \rightarrow \D (m+n) \\
  \inl (\varepsilon) & \mapsto (\varepsilon, 0) \\
  \inr (\varepsilon) & \mapsto (0, \varepsilon)
\end{align*}
We can also define a diagonal on the $1$-disk
\begin{align*}
  D : \D (1) & \rightarrow \D (2) \\
  \varepsilon & \mapsto (\varepsilon,\varepsilon)
\end{align*}

\begin{definition}\label{def:inf-lin}
  Let $(X,x)$ be a pointed type.
  $X$ is \notion{infinitesimally linear at $x$} if for every $m, n : \N$ the induced map
  \begin{align*}
    \inc_{m,n}^* : (X,x)^{\D (m+n)} \rightarrow (X,x)^{\D (m)} \times (X,x)^{\D (n)}
  \end{align*}
  is an equivalence.
\end{definition}

\begin{example}\label{ex:sch-inf-lin}
  By \cite{diffgeo-article}[Lemma 2.2.5], any scheme is infinitesimally linear at all of its points.
  In particular, the finite free $R$-modules $R^n$ are infinitesimally linear at any point.
\end{example}

Infinitesimal linearity is a local property in the following sense.

\begin{proposition}\label{prp:inf-lin-hood}
  Let $(X,x)$ be a pointed type.
  $X$ is infinitesimally linear at $x$ if and only if $\hood{X,x}{1}$ is infinitesimally linear at $x$.
\end{proposition}

\begin{proof}
  By \cref{prp:hood-factorization} any map $\D (n) \rightarrow_\ast (X,x)$ factors uniquely through $\hood{X,x}{1}$.
  Thus, we have the desired lifting property for $(X,x)$ if and only if we have it for $\hood{X,x}{1}$.
\end{proof}

\begin{lemma}\label{lem:projective-exact}
  Let $(X,x)$ be projective among types.
  The functor $(\argmhole,0)^{(X,x)} : \modcat{R} \rightarrow \modcat{R}$ is exact.
\end{lemma}

\begin{proof}
  Let $M \twoheadrightarrow N$ be an epimorphism of $R$-modules, and suppose we have a pointed map $f : (X,x) \rightarrow_\ast (N,0)$.
  We merely get a lift $\hat{g} : X \rightarrow M$ of $f$ along the epi.
  The map $g = \hat{g} - \hat{g}(0)$ is a pointed lift of $f$, as desired.
\end{proof}

\begin{proposition}\label{prp:tangent-exact}
  The functors $(\argmhole,0)^{\mathbb{D}(m+n)},\ (\argmhole,0)^{\mathbb{D}(m)} \times (\argmhole,0)^{\mathbb{D}(n)}$ are exact on $\modcat{R}$.
  In particular, $\tangent{}{0}$ is exact on $\modcat{R}$.
\end{proposition}

\begin{proof}
  Immediate by \cref{lem:disk-projective} and \cref{lem:projective-exact}.
\end{proof}

\begin{proposition}\label{prp:cok-ext-inf-lin}
  Infinitesimally linear $R$-modules at $0$ are closed under cokernels and extensions.
\end{proposition}

\begin{theorem}[\cite{etale-draft}{[Theorem 1.1.10]}]\label{thm:tangent-inf-lin}
  Let $X$ be an infinitesimally linear type at $x : X$.
  Then the tangent space $\tangent{X}{x}$ admits an $R$-module structure.
  This module structure will be called the \notion{microlinear module structure} on $\tangent{X}{x}$. \earlyqed
\end{theorem}

We can describe the relevant binary operations of the microlinear module structure.
Suppose we are given two tangent vectors $\upsilon, \nu : \tangent{M}{0}$, then $\upsilon \boxplus \nu$ (the microlinear addition) is defined by the composite $(\inc_{1,1}^\ast)^{-1}[\upsilon,\nu] \circ D$.
In other words, we obtain this map by lifting in the following diagram
\begin{center}
  \begin{tikzcd}
    & \D (1) \underset{\ast}{\coprod} \D (1) \ar{rd}{[\upsilon,\nu]} \ar{d}{\inc_{1,1}} \\
    \D (1) \ar{r}{D} \ar[bend right]{rr}{\upsilon \boxplus \nu} & \D (2) \ar[dashed]{r}{\exists !} & M
  \end{tikzcd}
\end{center}
The (microlinear) scalar multiplication by $r : R$ is defined by $(r \boxdot \upsilon)(\varepsilon) = \upsilon(r\varepsilon)$ for any $\varepsilon : \D (1)$.

\begin{proposition}[\cite{diffgeo-article}{[Lemma 2.2.6]}]\label{prp:derivative-inf-lin}
  Let $f : (X,x) \rightarrow_\ast (Y,y)$ be a pointed map between infinitesimally linear types.
  Then $\der{f}{x}$ is $R$-linear with respect to the microlinear module structure. \earlyqed
\end{proposition}

\begin{theorem}\label{thm:tangent-ptw-micro-add}
  Suppose that $M$ is an infinitesimally linear $R$-module at $0$.
  Then the pointwise and microlinear addition on $\tangent{M}{0}$ are the same.
\end{theorem}

\begin{proof}
  We will use Eckmann-Hilton to prove this theorem.
  It is clear that the identity for both of these $R$-module structures is the map constant on $0$, so it remains to argue that the additions are compatible.
  Let $+$ denote the addition on $M$, and we use the same symbol for the pointwise addition on $\tangent{M}{0}$.
  Let $\boxplus$ denote the microlinear addition on $\tangent{M}{0}$.
  Our goal is to show that for any $\alpha, \beta, \gamma, \delta : \tangent{M}{0}$, $(\alpha + \gamma) \boxplus (\beta + \delta) = (\alpha \boxplus \beta) + (\gamma \boxplus \delta)$ holds.

  The map $(\alpha + \gamma) \boxplus (\beta + \delta)$ is given by lifting in the following diagram
  \begin{center}
    \begin{tikzcd}
      & D (1) \underset{\ast}{\coprod} D (1) \ar{d}{\inc_{1,1}} \ar{r}{[\Delta,\Delta]} & D (1)^2 \underset{*}{\coprod} D (1)^2 \ar{rr}{[\alpha \times \gamma, \beta \times \delta]} & & M^2 \ar{d}{+} \\
      D (1) \ar{r}{D} \ar[bend right, looseness = 0.5]{rrrr}{(\alpha + \gamma) \boxplus (\beta + \delta)} & D (2) \ar[dashed]{rrr}{\exists !} & & & M
    \end{tikzcd}
  \end{center}
  We get $(\alpha \boxplus \beta) + (\gamma \boxplus \delta)$ by lifting in the following diagram
  \begin{center}
    \begin{tikzcd}
      & & (D (1) \underset{*}{\coprod} D (1))^2 \ar{d}{\inc_{1,1}\times \inc_{1,1}} \ar[bend left]{rd}{[\alpha,\beta] \times [\gamma,\delta]} \\
      D (1) \ar{r}{\Delta} \ar[bend right, looseness = 0.5]{rrrr}{(\alpha \boxplus \beta) + (\gamma \boxplus \delta)} & D (1)^2 \ar{r}{D \times D} & D (2)^2 \ar[dashed]{r}{\exists} & M^2 \ar{r}{+} & M
    \end{tikzcd}
  \end{center}
  By using the naturality square involving $\Delta$ and $D$ we can glue the lower parts of the diagrams together.
  \begin{center}
    \begin{tikzcd}
      D (1) \ar{r}{D} \ar{d}{\Delta} & D (2) \ar[dashed, bend left, looseness = 0.5]{rrd}{\exists !} \ar{d}{\Delta}\\
      D (1)^2 \ar{r}{D \times D} & D (2)^2 \ar[dashed]{r}{\exists} & M^2 \ar{r}{+} & M
    \end{tikzcd}
  \end{center}
  The maps $(\alpha + \gamma) \boxplus (\beta + \delta)$ and $(\alpha \boxplus \beta) + (\gamma \boxplus \delta)$ are equal if the triangle on the right above commutes.
  Since $M$ is infinitesimally linear, this is the case iff pre-composing with $\inc_{1,1}$ makes the triangle commute.
  Using the naturality square involving $\Delta$ and $\inc_{1,1}$, this happens if the outer arrows of the following diagram commute
  \begin{center}
    \begin{tikzcd}
      D (1) \underset{*}{\coprod} D (1) \ar{r}{\Delta} \ar{d}{\inc_{1,1}} & (D (1) \underset{\ast}{\coprod} D (1))^2 \ar{d}{\inc_{1,1}\times \inc_{1,1}} \ar[bend left]{rd}{[\alpha,\beta] \times [\gamma,\delta]} \\
      D (2) \ar{r}{\Delta} \ar[dashed, bend right, looseness = 0.5]{rrr}{\exists !} & D (2)^2 \ar[dashed]{r}{\exists} & M^2 \ar{r}{+} & M
    \end{tikzcd}
  \end{center}
  One can now observe that the outer arrows commute because the following square does
  \begin{center}
    \begin{tikzcd}
      D (1) \underset{*}{\coprod} D (1) \ar{r}{[\Delta, \Delta]} \ar{d}{\Delta} & (D (1) \underset{*}{\coprod} D (1))^2 \ar{d}{[\alpha, \beta]\times [\gamma,\delta]} \\
      D (1)^2 \underset{*}{\coprod} D (1)^2 \ar{r}{[\alpha\times \gamma, \beta\times \delta]} & M^2
    \end{tikzcd}
  \end{center}
  From this we conclude that $(\alpha + \gamma) \boxplus (\beta + \delta) = (\alpha \boxplus \beta) + (\gamma \boxplus \delta)$ holds, and the binary operations are the same.
\end{proof}

\begin{remark}\label{rem:ptw-micro-scalar}
  In general, the scalar multiplication induced by the pointwise and microlinear module structures differs.
  Let $\nu : \tangent{M}{0}$ be a tangent vector on $M$, $r : R$, and $\varepsilon : \D (1)$.
  Then $(r\boxdot \nu)(\varepsilon) = \nu(r\varepsilon)$ and $(r \cdot \nu)(\varepsilon) = r\nu(\varepsilon)$.
  We can observe that $r\boxdot\nu = r\cdot \nu$ only whenever $\nu$ respects scalar multiplication.
\end{remark}

\begin{corollary}\label{cor:can-nat-inf-lin}
  Let $M$ be an infinitesimally linear $R$-module at $0$.
  Then the map ${\can : M \rightarrow \tangent{M}{0}}$ is $R$-linear with respect to the microlinear module structure on $\mathrm{T}_0M$.
  % In particular, the naturality square below
  % \begin{center}
  %   \begin{tikzcd}
  %     M \ar{d}{f} \ar{r}{\can} & \tangent{M}{0} \ar{d}{\der{f}{0}} \\
  %     N \ar{r}{\can} & \tangent{N}{0}
  %   \end{tikzcd}
  % \end{center}
  % is a square of $R$-modules with $R$-linear maps whenever $M$ and $N$ are infinitesimally linear $R$-modules at $0$ and $f$ is an $R$-linear homomorphism.
\end{corollary}

\begin{proof}
  The map $\can$ is $R$-linear when we equip $\tangent{M}{0}$ with the pointwise module structure.
  Thus, by \cref{thm:tangent-ptw-micro-add} the map $\can$ respects the microlinear addition.
  It remains to show that $\can$ respects the microlinear scalar multiplication.
  Let $m : M$, $\varepsilon : \mathbb{D}(1)$ and $r : R$, then we compute $(r\boxdot\mathrm{can}(m))(\varepsilon) = \mathrm{can}(m)(r\varepsilon) = r\varepsilon m = \varepsilon rm = \mathrm{can}(rm)(\varepsilon)$.
\end{proof}

We collect the definitions developed in this section under a single umbrella.

\begin{definition}\label{def:tangent-like-module}
  \rednote{This name should be debated: tangent-like vs. linearizing?}
  Let $M$ be an $R$-module.
  We say that $M$ is \notion{tangent-like} if it is both infinitesimally linear at $0$ and tangential.
\end{definition}

\begin{proposition}\label{prp:lim-col-ext-tangent-like}
  Tangent-like $R$-modules are closed under biproducts, kernels, cokernels, extensions, and exponentials.
\end{proposition}

\begin{proof}
  This follows from \cref{prp:exp-tangential}, \cref{prp:lim-inf-lin}, \cref{prp:cok-ext-inf-lin}, and \cref{prp:lim-col-ext-tangential}.
\end{proof}

\begin{example}\label{ex:tangent-like-module}
  The $R$-modules $R^n$, $R[x]$, and $R[[x]]$ are tangent-like.
\end{example}

Tangent-like $R$-modules admit a powerful linearization trick.

\begin{theorem}\label{thm:linearize-lift}
  Suppose we have the following square of $R$-modules
  \begin{center}
    \begin{tikzcd}
      A \ar{r}{u} \ar{d}{m} & X \ar{d}{e} \\
      B \ar{r}{v} & Y
    \end{tikzcd}
  \end{center}
  where $A$ and $Y$ are tangential, and $B$ and $X$ are tangent-like.
  If there merely exists a non-linear but pointed solution $h':(B,0) \rightarrow_\ast (X,0)$ to the lifting problem above, then there merely exists a linear solution as well.
\end{theorem}

\begin{proof}
  Assume we have a non-linear solution $h':B \rightarrow X$.
  By \Cref{prp:derivative-inf-lin}, $\der{h'}{0}$ is a linear map, so we define the linear solution as $h = \can^{-1} \circ \der{h'}{0} \circ \can$.
  To verify that $h$ is a solution, we will use functoriality of taking tangent spaces, as we have the following equalities: $\der{h'}{0} \circ \der{m}{0} = \der{u}{0}$, and $\der{e}{0} \circ \der{h'}{0} = \der{v}{0}$.
  Using the naturality of $\can$ (\cref{prp:can-nat}) and functoriality of taking tangent spaces, we can compute the following:
  \rednote{Can this be prettier?}
  \begin{equation*}
    h \circ m = \can^{-1} \circ \der{h'}{0} \circ \can \circ m = \can^{-1} \circ \der{h'}{0} \circ \der{m}{0} \circ \can = \can^{-1} \circ \der{u}{0} \circ \can = u
  \end{equation*}
  \begin{equation*}
    e \circ h = e \circ \can^{-1} \circ \der{h'}{0} \circ \can = \can^{-1} \circ \der{e}{0} \circ \der{h'}{0} \circ \can = \can^{-1} \circ \der{v}{0} \circ \can = v
  \end{equation*}
  Thus, $h$ is a linear solution as desired.
\end{proof}
