\begin{lemma}
  For any entourage $E$ of $\mathbb N\infty$, there exists a cofinite set $C$ such that $C \times C \subseteq E$. 
\end{lemma} 
\begin{proof}
  By the basis theorem for products and the observation that $(\infty,\infty) \in E$, there must be some open subset $U\subseteq \Noo$ such that $(\infty, \infty) \in U \times U \subseteq E$. And as the decidables form a basis for a Stone space, there is some decidable $D\subseteq \Noo$ containing $\infty$ with $D\subseteq U$. And as $\infty \in D$, $D$ is cofinite, and $D\times D \subseteq E$.
\end{proof}

\begin{definition}
  Let $Q\subseteq S$ be any subset of a Stone space.
  And let $f : Q \to T$ be any map into another Stone space. 
  Then $f$ is called \textbf{relatively uniformly continuous} if for all entourages $E$ of $T$, there exists an entourage $F$ of $S$ such that whenever $x \in F \cap Q$, we have $f(x) \in E$. 
\end{definition}
\begin{conjecture}
  Any relatively uniformly continuous function from a dense subset of a Stone space can be extended to a function.
