\begin{definition}
  A type $X$ is 
  \begin{itemize}
    \item \textbf{overt} if for all $U\subseteq X$ open, the statement $\exists_{x:X} U(x)$ is open
    \item \textbf{compact} if for all $U\subseteq X$ open, the statement $\forall_{x:X} U(x)$ is open. 
    \item \textbf{condensed} if for all $C\subseteq X$ closed, the statement $\exists_{x:X} C(x)$ is closed. 
    \item \textbf{discrete} if equality is open. 
    \item \textbf{Hausdorff} if equality is closed. 
  \end{itemize}
\end{definition}
Recall that compact Hausdorff spaces are compact, Hausdorff and condensed, and overtly discrete spaces overt and discrete. 

\begin{definition}
  A subset $K\subseteq X$ is \textbf{subcompact} if that for all open subsets $U\subseteq X$ we have that $K\subseteq U$ open. 
  Denote $Comp(X)$ for the subcompact spaces of $X$. 
\end{definition}
\begin{remark}
  Closed subspaces of compact Hausdorff spaces are subcompact. 
\end{remark}

\begin{definition}
  Let $X$ be a type. A \textbf{local compactness structure} on $X$ consists of a map 
  $b = (b_O,b_K) : \mathcal B \to ((X \to \Open) \times (Comp(X)))$ such that:
  \begin{itemize}
    \item $\mathcal B$ is overt. 
    \item For all $j:\mathcal B$, $b_O(j) \subseteq b_K(j)$. 
    \item For all $U\subseteq X$ open and $x\in U$, there exists some $j\in\mathcal B$ such that $x \in b_O(j)\subseteq b_K(j) \subseteq U$. 
  \end{itemize}
\end{definition}
