\begin{definition}

\subsection{Remarks on invertible modules}

%A key property we will need is that given $A,B:R-\Alg_{nc}$ that are finite free as $R$-module, $\Mor(A,B)$ is an \'etale sheaf.

\begin{lemma}
The type $\B R^\times$ is an \'etale sheaf.
\end{lemma}

\begin{proof}
Because $\Mod{R}{ff}$ is an \'etale sheaves and the rank is decidable.
\end{proof}

\begin{lemma}\label{invertible-R-module-free-rank-1}
Let $L$ be an $R$-module. TFAE:
\begin{enumerate}[(i)]
\item $L$ is free of rank $1$.
\item The map $L\otimes L^*\to R$ is an equivalence.
\item $L$ is invertible.
\end{enumerate}
\end{lemma}

\begin{proof}
The implications $(i)\Rightarrow (ii)\Rightarrow(iii)$ are obvious. For $(iii)\Rightarrow (i)$, assume $L$ invertible in $\Mod{R}{}$, by \Cref{morita-main-theorem} we have that $L$ is finitely generated projective and therefore finite free, and its rank has to be $1$ by dimension computation. 

Now we give an alternative elementary proof of $(iii)\Rightarrow (i)$.

Assume $M$ invertible in $\Mod{R}{}$, then we have $N$ with $\alpha:M\otimes N = R$. This gives us $\sum_i m_i\otimes n_i$ such that $\sum_i \alpha(m_i\otimes n_i) = 1$, which by locality and rescaling give us $m:M$ and $n:N$ such that $\psi(m\otimes n) = 1$. Then using the map $\bar{\psi} N\to M^*$ we get an element $f:M^*$ such that $f(x) = 1$.

Then $f:M^*$ is clearly surjective, let us prove it is injective. Assume $y:M$ such that $f(y) = 0$, then $y\otimes m = 0$ and therefore $y\otimes f = 0$. Then consider the map $M\otimes M^*\to M$ sending $y\otimes f$ to $z\mapsto f(z)y$, since $y\otimes f = 0$ we get $f(z)y = 0$ for all $z:R$. For $z=x$ we get $y=0$ as we wanted.
\end{proof}

\begin{lemma}\label{invertible-azumaya-finite-free}
Given $A,B:\Az$, any invertible in $\Mod{A}{B}$ is finite free as an $R$-module.
\end{lemma}

\begin{proof}
By \Cref{morita-main-theorem} we have that such an $M$ is a finitely generated projective $A$-module. Then it is a direct summand of $A^n$, so a direct summand of a finite free $R$-module and we can conclude.
\end{proof}

\begin{lemma}\label{azumaya-tensor-finitely-presented}
Assume $A:\Az$ with $M:\Mod{}{A}$ and $N:\Mod{A}{}$ both finite free as $R$-module. Then $M\otimes_AN$ is finitely presented as an $R$-module. In particular it is an \'etale sheaf.
\end{lemma}

\begin{proof}
Consider $(a_i)_{i:I}$ (resp.\ $(m_k)_{k:K}$, $(n_l)_{l:L}$) a basis of $A$ (resp.\ $M$, $N$) as an $R$-module. Then $M\otimes_A N = M\otimes_RN / (m_ka_i\otimes n_l - m_k\otimes a_in_l)_{i:I,k:K,l:L}$ as an $R$-module.
\end{proof}

\begin{corollary}\label{morita-context-etale-sheaf}
Given $A,B:\Az$, the type $\Mor(A,B)$ is an \'etale sheaf.
\end{corollary}

\begin{proof}
By \Cref{invertible-azumaya-finite-free}, we have that $\Mor(A,B)$ is equivelent to giving $M$ finite free $R$-module with a $A\otimes B^{op}$ actions and $N,N'$ finite free $R$-module with $B\otimes A^{op}$-actions with $M\otimes_AN = B$ and $N'\otimes_BM = A$.

We know the type of finite free $R$-module is an \'etale sheaf, so we just need $M\otimes_AN$ and $N'\otimes_BM$ to be \'etale sheaves. This is \Cref{azumaya-tensor-finitely-presented}.
\end{proof}


\subsection{Azumaya algebras and \'etale torsors}

\begin{definition}
Given $A:\Az$, we define $\psi(A) = \Sigma_{V:\Mod{R}{ff\geq 1}}\, A=\End_R(V)\}$.
\end{definition}

\begin{lemma}\label{psi-equivalent-mor}
Given $A:\Az$, we have $\psi(A) = \Mor(A,R)$.
\end{lemma}

\begin{proof}
This is a direct application of \Cref{morita-main-theorem}. Perhaps there is a simpler proof for this specific case.
\end{proof}

Given $Y$ a pointed type, recall that a type $X$ is an \'etale $Y$-torsor if:
\begin{itemize}
\item It is a $Y$-pseudotorsor, i.e.\ for all $x:X$ we have an equivalence $(X,x) = (Y,*)$. Beware that this is a structure and not a proposition,
\item $X$ is an \'etale sheaf and we have $\propTrunc{X}_{\Et}$.
\end{itemize}

\begin{corollary}
Given $A:\Az$, the type $\psi(A)$ is an \'etale $\B R^\times$-torsor.
\end{corollary}

\begin{proof}
By \Cref{psi-equivalent-mor}, we have that $\psi(A)$ is equivalent to $\Mor(A,R)$, which is a $\Mor(R,R)$-pseudotorsor since Morita equivalences compose in the expected way. Then by \Cref{invertible-R-module-free-rank-1} we have $\Mor(R,R) = \B R^\times$, so we get a $\B R^\times$-pseudotorsor. 

Now $\psi(A)$ is an \'etale sheaf, as $A$ and the type of finite free module are an \'etale sheaves. When checking $\propTrunc{\psi(A)}_{\Et}$ we can assume $A=M_{n+1}(R)$ for some $n:\N$, and then $V=R^{n+1}$ works.
\end{proof}

\begin{corollary}
We have a fiber sequence of pointed types:
\[\B R^\times \to \{M\ \text{finite\ free\ of\ rank}\ \geq 1\} \to \Az\to \B_{\Et}^2 R^\times\]
\end{corollary}

\begin{proof}
The fiber of $\psi:\Az\to \B_{\Et}\B R^\times$ over $*$ is $\Sigma_{A:\Az}\psi(A)=*$, which in turn is equivalent to $\Sigma_{A:\Az}\psi(A)$ as proving a torsor trivial is the same as giving a point in it.  By the singleton principle this the expected type. Then we can extend the sequence by $\Omega \B_{\Et}\B R^\times$, which is indeed $\B R^\times$.
\end{proof}

\begin{lemma}\label{torsor-morita-invariant}
Given $A,B:\Az$, we have an equivalence $\Mor(A,B) \simeq (\psi(A) = \psi(B))$ sending the composition of Morita equivalences to composition of path and is an equivalence. 
\end{lemma}

\begin{proof}
Using that $\psi(A) = \Mor(A,R)$, the fact that Morita equivalences compose appropriately gives us the map commuting with composition. To check this is an equivalence, by \Cref{morita-context-etale-sheaf} we can assume that $A$ and $B$ are matrices algebras, then using \Cref{matrix-morita-equivalence} and the commutative diagram
\[\begin{tikzcd}
\Mor(M_k(R),M_l(R)) \ar[r]\ar[d]& \psi(M_k(R)) = \psi(M_l(R))\ar[d] \\
\Mor(R,R)\ar[r] & \psi(R)=\psi(R)\\
\end{tikzcd}\]
where vertical maps are equivalences, it is enough to check the the bottom map is an equivalence. We even have the more general property that the map $\Mor(A,R) \to \psi(A)=\psi(R)$ is an equivalence simply because for any pseudotorsor $X$ we have that $X=(X=*)$. We omit the verification that the generic such equivalence indeed match with the $\psi$ we defined.
\end{proof}

\begin{remark}
Using this we can even prove that we have an equality $\Az/\Mor = \B^2_{\Et}R^\times$, where the quotient $\Az/\Mor$ happens in \'etale sheaves. I don't know whether this holds with the usual quotient, or equivalently whether the naive $\Az/\Mor$ is an \'etale sheaf.
\end{remark}
