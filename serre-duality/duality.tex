We prove Serre-duality. For now only for vector bundles.


\begin{lemma}
  \label{bundle-homs-to-dual-tensor}
  Let $X$ be a type and $\mathcal L:X\to \Line$ a line bundle.
  For all $\mathcal F:X\to\Mod{R}$ we have an isomorphism of $R$-module bundles:
  \[ \HomBun_{\Mod{R}(X)}(\mathcal L,\mathcal F) \to \mathcal L^{\vee} \otimes \mathcal F\rlap{.}\]
  (where $\HomBun_{\Mod{R}(X)}(\mathcal L,\mathcal F)= (x:X) \mapsto \Hom_{\Mod{R}}(\mathcal L_x,\mathcal F_x)$)
  In particular, we have
  \[\Hom_{\Mod{R}(X)}(\mathcal L,\mathcal F) \simeq H^0(X,\mathcal L^{\vee} \otimes \mathcal F)\rlap{.} \]
\end{lemma}

\begin{proof}
  Let $x:X$ and choose an isomorphism of $R$-modules $\mu:\mathcal L_x\to R$.
  Note first, that we have an isomorphism
  \[ \psi: \Hom_{\Mod{R}}(R,\mathcal F_x) \to R^{\vee}\otimes \mathcal F_x\]
  given by mapping $\varphi:\Hom_{\Mod{R}}(R,\mathcal F_x)$ to $\id_R\otimes \varphi(1)$.
  So
  \[ \mu^{\vee}\otimes \id_R \circ \psi\circ (\_\circ \mu^{-1})
  : \Hom_{\Mod{R}}(\mathcal L_x,\mathcal F_x) \to {\mathcal L_x}^{\vee}\otimes \mathcal F_x\]
  is a map of the desired type and by elementary calculation using linearity it is independent of the choice of $\mu$.
\end{proof}

\begin{theorem}[Duality for $\bP^n$]
  Let $X :\equiv \bP^n$ and let $\mathcal E : X \to \Mod{R}_{\mathrm{fp}}$ be a finitely presented bundle.
  \begin{enumerate}
    \item[i)] $H^n(\bP^n,\omega)=R$.
    \item[ii)] There is a natural perfect pairing
    \[
      \Hom_X(\mathcal E,\omega) \times H^n(X,\mathcal E) \to H^n(X,\omega)
    \]
    inducing a natural isomorphism $ \theta^0(\mathcal E) : \Hom_X(\mathcal E,\omega) \to H^n(X,\mathcal E)^\vee$.
  \item[iii)] If $\mathcal E$ is even a vector bundle then for every $i\geq 0$ there is a natural functorial isomorphism
    \[
      \theta^i(\mathcal E) : H^i(X, \mathcal E^\vee \otimes \omega) \to H^{n-i}(X,\mathcal E)^\vee
    \]
  \end{enumerate}
\end{theorem}

\begin{proof}
  \begin{enumerate}
    \item[i)] This follows from \Cref{canonical-bundle-as-twist}.
    \item[ii)] The pairing is given by letting $\varphi:\Hom_X(\mathcal F,\omega)$ induce a $\varphi^*:H^n(X,\mathcal F)\to H^n(X,\omega)$.
    In the case $\mathcal F\simeq \mathcal O(q)$ we compute $\Hom_X(\mathcal O(q),\omega)\simeq H^0(\bP^n,\omega\otimes \mathcal O(-q))$.
    So in this case, we can conclude as we know the cohomology of twisting sheaves.
    By \ref{fp-module-bundle-globally-generated}, we get an exact sequence:
    \[ \mathcal O(p)^{\oplus n}\to\mathcal O(p')^{\oplus n'}\to \mathcal F\to 0 \]
    and the functors $\Hom_X(\_,\omega)$ and $H^n(X,\_)^\vee$ both map right exact sequences to left exact sequences. So we can conclude with the 5-lemma.
  \item[iii)] Since vector bundles have finitely adequate cohomology by \Cref{vector-bundles-have-faq-cohomology-which-eventually-vanishes} we know that both $H^i(X, (\_)^\vee \otimes \omega)$ and $H^{n-i}(X,\_)^\vee$ are contravariant and exact $\delta$-functors.
    These are also both coherently coeffaceable by \Cref{vector-bundles-can-be-coherently-resolved} and thus both universal by \Cref{coherently-effaceable-implies-universal}.
  \end{enumerate}
\end{proof}
