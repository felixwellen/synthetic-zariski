\section{SSD, Stone spaces and compact Hausdorff spaces}


\frame{\frametitle{Stone spaces}

\bloc{Definition}{
A type $X$ is a \red{Stone space} if it is a sequential limit of finite types.
}

\frame{\frametitle{Compact Hausdorff spaces}

Stone spaces are not stable under quotients.

\bloc{Definition}{
A set $X$ is a \red{compact Hausdorff space} if: 
\begin{itemize}
\item Its identity types are Stone spaces.
\item There exists $S:\Stone$ and $S\twoheadrightarrow X$.
\end{itemize}
}
\pause
\bloc{Example: \blue{The unit interval}}{
The type $\I = [0,1]$ is a compact Hausdorff space.
}

Indeed $\I$ is a quotient of $2^\N$.

\section{Intoduction to cohomology in HoTT}

\frame{\frametitle{Delooping abelian groups}

Fix $A$ an abelian group. We define $K(A,0)=A$.

\bloc{Proposition}{
Given $n>0$, there is a \blue{unique} pointed type $K(A,n)$ such that:
\begin{itemize}
\item $K(A,n)$ is $(n\,\hbox{-}1)$-connected and $n$-truncated.
\item $\Omega^n K(A,n) = A$.
\end{itemize}
}

$K(A,n)$ is called the \red{$n$-th delooping of $A$}.

}

\frame{\frametitle{Cohomology groups}

\bloc{Definition: \blue{Cohomology}}{
Given $n:\N,\, X:\Type$ and $A:X\to \Ab$, we define
\vspace{0.2cm}
\[H^n(X,A) = \propTrunc{\Pi_{x:X}\, K(A_x,n)}_0.\]
}

\frame{\frametitle{Tychonov and its dual}

We prove Barton-Commelin's condensed type theory axioms.
\pause
\bloc{Lemma: \blue{Tychonov}}{
If $I:\ODisc$ and $X:I\to \CHaus$, then $\Pi_{i:I}\, X_i$ is compact Hausdorff.
}
\pause
\bloc{Proposition: \blue{Tychonov's dual}}{
If $X:\CHaus$ and $I:X\to \ODisc$, then $\Pi_{x:X}\, I_x$ is overtly discrete.
}

This is encouraging. We have better!

}

\section{Cohomology of Stone and compact Hausdorff spaces}

\frame{\frametitle{\v{C}ech cohomology}

\bloc{Definition}{
A \v{C}ech cover consists of $X:\CHaus$ and $S:\Stone$ with $p:S\twoheadrightarrow X$.
}

\bloc{Definition}{
Given a \v{C}ech cover $p:S\twoheadrightarrow X$ and $A:X\to \Ab_{cp}$, we define $\check{H}^n(X,S,A)$ as the $n$-th cohomology group of
\vspace{0.2cm}
\[\Pi_{x:X}\, A_x^{S_x}\to \Pi_{x:X}\, A_x^{S_x^2} \to \Pi_{x:X}\, A_x^{S_x^3}\to \cdots.\]
}

$\check{H}^n(X,S,A)$ is called the \red{$n$-th \v{C}ech cohomology group of $X$ with coefficient in $A$}.

\frame{\frametitle{Main results}

\bloc{Theorem: \blue{Cohomology vanishing for Stone spaces}}{
Given $n>0$, $S:\Stone$ and $A:S\to\Ab_{cp}$, we have that
\vspace{0.2cm}
\[H^n(S,A) = 0.\]
}

\bloc{Theorem: \blue{\v{C}ech and regular cohomology agree on $\CHaus$}}{
Given a \v{C}ech cover $p:S\twoheadrightarrow X$ and $A:X\to\Ab_{cp}$, we have that
\vspace{0.2cm}
\[H^n(X,A) = \check{H}^n(X,S,A).\]
}

}

\frame{\frametitle{Applications}

\bloc{Lemma: \blue{Cohomology of the interval}}{
For $A:\Ab_{cp}$, we have that
\vspace{0.2cm}
\[H^n(\mathbb{I},A) = 
\begin{cases}
	A& $if$\ n=0 \\
    	0 & $otherwise.$
\end{cases}\]
}

%This means $\I$-localisation is well-behaved. For example:

\bloc{Lemma: \blue{Cohomology of the spheres}}{
For $\mathbb{S}^k = \{x_0,\hdots,x_k:\R\ |\ \Sigma_i\, x_i^2 = 1\}$ and $A:\Ab_{cp}$, we have that
\vspace{0.2cm}
\[H^n(\mathbb{S}^k,A) = 
\begin{cases}
	A & $if$\ n=0\ $or$\ n=k \\
    	0 & $otherwise.$
\end{cases}\]
}

This extends to all countable topological CW complex.

}
