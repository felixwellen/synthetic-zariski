 \section{Introduction to synthetic geometry}
 
 
 \frame{\frametitle{Synthetic geometries}
 
Say we want to work with some geometric object.

(e.g.\ differential manifolds, sequential spaces, schemes...)
 
 \bloc{Core idea (type theoretic version)}{
 \begin{enumerate}
 \item Embeds these objects in a suitable topos.
 \item Interpret MLTT + UIP in this topos.
\item Look for new axioms \red{(and modalities)} validated by this interpretation.
\item Work in MLTT + UIP + these axioms \red{(+ modalities)}.
\end{enumerate}
 }
 
 }
 
 
  \frame{\frametitle{What should we expect?}
 
 The objects of interest are types satisfying some proposition.
 
 \bloc{Slogan}{
 \begin{itemize}
 \item All types come with a topology/differential structure/etc.
 \item All maps are continuous/differentiable/etc.
 \end{itemize}
 }

This means we have to abandon LEM and AC.

 \section{Presheaf models}
 
 
 \frame{\frametitle{Goal}
 
 Assume given an algebraic theory $\T$, giving the local model for the space we want to study. 
 
For SAG we take $\T$ to be the theory of rings.
 
 \bloc{Goal}{
 We want to work synthetically in the classifying (higher) topos for $\T$.
 
 This is the topos of presheaf over $Alg_{\T}^{f.p.}$.
 }
 
 Idea: representables are dual to the algebras, and therefore model the affines.
 
 }
 
 
 \frame{\frametitle{The generic object}
 
 We have a presheaf sending any $A:\T\mhyphen Alg_{f.p.}$ to its underlying set.
 
 It is a $\T$-algebra in $PSh(\T\mhyphen Alg_{f.p.})$.
 
 \bloc{Axiom 1 (Generic object)}{
 There is a set $\G$ with a $\T$-algebra.
 }
 
 }
 
 
 \frame{\frametitle{Representables}
 
 \bloc{Definition}{
 A $\G$-algebra is a $\T$-algebra $A$ with a morphism $\G\to A$
 }
 
 \bloc{Definition}{
 Given $A:\G\mhyphen-Alg_{f.p.}$, we define:
 \[\Spec(A) = \Hom_\G(A,\G)\]
 }
 
 \bloc{Example (SAG)}{
 We have a ring $R$. Then:
 \[\Spec(R[X]) = \Hom_R(R[X],R) = R\]
 Given $P:R[X]$, then:
 \[\Spec(R[X]/P) = \Hom_{R\mhyphen Alg}(R[X]/P,R) = \{x:R\ |\ P(x)=0\}\]
 }
 
 }
 
 
 \frame{\frametitle{Duality axiom}
 
 \bloc{Axiom 2 (Duality)}{
 Assume given $A:\G\mhyphen Alg_{f.p.}$, then the map:
 \[ A \to R^{\Spec(A)}\]
 is an equivalence.
 }
 
 \bloc{Corollary}{
 The map:
 \[\Spec : \G\mhyphen Alg_{f.p.}\to \Set\]
 is an an embedding.
 }
 
 The set in the image of $\Spec$ are called the representables. In SAG they are called affine schemes
 
 \bloc{Corollary}{
 Given $A,B:\G\mhyphen Alg_{f.p.}$, we have that:
 \[(\Spec(B)\to \Spec(A)) = \Hom_{\G\mhyphen Alg}(A,B)\]
 }
 
 \bloc{Example (SAG)}{
 \[(R \to R) = \Hom_{R\mhyphen Alg}(R[X],R[X]) = R[X]\]
 }
 
 Kind of like Yoneda I guess?
 
 }
 
 
 \frame{\frametitle{Choice for representable}
 
 We know that representable in $PSh(\T\mhyphen Alg)$ are projective.
 
 \bloc{Axiom 3 (Choice for representable)}{
 Given $A:\G\mhyphen Alg_{f.p.}$, the type $\Spec(A)$ has choice.
 }
 
 Here we can already say some relatively interesting stuff.
 
 \bloc{Corollary}{
 Given $A:\G\mhyphen Alg_{f.p.}$, the map:
 \[(A\mhyphen Alg_{f.p.}) \to (\Spec(A)\to \G\mhyphen Alg_{f.p.})\]
 TODO does this even make sense?
 }
 
 }
 
 
 \frame{\frametitle{MAYBE Digression on qc modules}
 
 }
 
 
 \frame{\frametitle{$\N$ is constant}
 
 We interpret the type $\N$ as the constant presheaf sending $A:\G\mhyphen Alg_{f.p.}$ to $\N$.
 
 \bloc{Axiom bonus ($\N$ constant)}{
 For any $A:\G\mhyphen Alg_{f.p.}$, the map
 \[\N\to \N^{\Spec(A)}\]
 is an equivalence.
 }
 
 }
 
 
 \frame{\frametitle{The presheaf axioms}
 
 \bloc{Theorem [Jonas etc...]}{
 Given $\T$ an algebraic theory, there exists a model of HoTT plus:
 \begin{itemize}
 \item There is $\G$ a $\T$-algebra.
 \item For all $A:\G\mhyphen Alg_{f.p.}$, the map $A\to \G^{\Spec(A)}$ is an equivalence.
 \item For all $A:\G\mhyphen Alg_{f.p.}$, the set $\Spec(A)$ has choice.
 \end{itemize}
 }
 
 }
