 \section{Introduction to synthetic geometry}
 
 
 \frame{\frametitle{Synthetic geometry}
 
Say we want to work with some geometric objects,

e.g.\ differential manifolds, sequential spaces, schemes...
 
 \bloc{Core idea (an ahistorical type-theoretic reframing)}{
 \begin{enumerate}
 \item Embeds objects of interests in a topos.
 \item Interpret MLTT + UIP in this topos.
\item Look for new axioms \red{(and modalities)} validated by this interpretation.
\item Work in MLTT + UIP + these axioms \red{(+ modalities)}.
\end{enumerate}
 }
 
 }
 
 
  \frame{\frametitle{What should we expect?}
 
 The objects of interest are types satisfying some proposition.
 
 \bloc{Slogan}{
 \begin{itemize}
 \item All types come with a topology/differential structure/etc.
 \item All maps are continuous/differentiable/etc.
 \end{itemize}
 }

This means we have to abandon LEM and AC.
 
 }

 \frame{\frametitle{Synthetic homotopy theory}
 
 Say we want to work with homotopy types / $\infty$-groupoids.
 
\bloc{Core idea}{
\begin{enumerate}
\item MLTT can be interpreted in $\infty$-groupoids.
\item This validates HoTT (= univalence + HITs).
\item Work in HoTT.
\end{enumerate}
}
 
 }
 
 
 \frame{\frametitle{We can do both!}
 
 \bloc{Core idea}{
 \begin{enumerate}
 \item Embeds objects of interest in an $\infty$-topos.
 \item Interpret HoTT in this topos.
\item Look for new axioms \red{(and modalities)} validated by this interpretation.
\item Work in HoTT + these axioms \red{(+ modalities)}.
\end{enumerate}
 }

 \bloc{Why do this?}{
 \begin{itemize}
 \item Objects of interests have $\infty$-groupoid-like structure.
 \item We want to use some homotopy theory (e.g.\ cohomology).
 \item For the convenience of univalence and HITs.
 \end{itemize}
 }
 
 }

 \section{Presheaf models}
 
 
 \frame{\frametitle{Goal}
 
This section present results from [Coquand, Höfer, Sattler 26].
 
 Assume given an algebraic theory $\T$, modelling our building blocks. 
 
For SAG, $\T$ is the theory of commutative rings.
 
 \bloc{Goal}{
 Work synthetically in the classifying $\infty$-topos for $\T$.
 }
 
 This is the topos of presheaf over $\T\mhyphen\Alg_{f.p.}$.
 
 \bloc{Idea}{
 Representables are dual to the algebras, and therefore model the basic building blocks.
 }
 
 }
 
 
 \frame{\frametitle{The generic object}
 
 We have a presheaf sending any $A:\T\mhyphen\Alg_{f.p.}$ to its underlying set.
 
 It is a $\T$-algebra in $\PSh(\T\mhyphen\Alg_{f.p.})$.
 
 \bloc{Axiom 1 (Generic object)}{
 There is a set $\G$ with a $\T$-algebra structure.
 }
 
 }
 
 
 \frame{\frametitle{The spectrum construction}
 
 \bloc{Definition}{
 A $\G$-algebra is a $\T$-algebra $A$ with a morphism from $\G$ to $A$.
 }
 
 \bloc{Definition}{
 Given $A:\G\mhyphen\Alg_{f.p.}$, we define
 
 \[\Spec(A) = \Hom_{\G\mhyphen\Alg}(A,\G).\]
 }
 
 \bloc{Example (SAG)}{
% We have a ring $R$. Then
 
% \[\Spec(R[X]) = \Hom_R(R[X],R) = R.\]
 Given $P:R[X]$, then
 
 \[\Spec(R[X]/P) = \Hom_{R\mhyphen Alg}(R[X]/P,R) = \{x:R\ |\ P(x)=0\}.\]
 }
 
 }
 
 
 \frame{\frametitle{Duality axiom and representable}
 
 \bloc{Axiom 2 (Duality)}{
 Assume given $A:\G\mhyphen\Alg_{f.p.}$, then the map
 
 \[A \to R^{\Spec(A)}\]
 is an equivalence.
 }
 
 \bloc{Corollary}{
 The map
 
 \[\Spec : \G\mhyphen\Alg_{f.p.}\to \Set\]
 is an an embedding.
 }
 
The sets in the image of $\Spec$ are called the affines. 
 
In SAG they are called affine schemes.
 
 }
 
 
 \frame{\frametitle{Duality: easy consequences}
 
  \bloc{Corollary}{
 Given $A,B:\G\mhyphen Alg_{f.p.}$, we have that
 
 \[(\Spec(B)\to \Spec(A)) = \Hom_{\G\mhyphen\Alg}(A,B).\]
 }
 
 \bloc{Example (SAG)}{
 \[(R \to R) = \Hom_{R\mhyphen\Alg}(R[X],R[X]) = R[X]\]
 }
 
 }

 \section{Sheaf models from internal topologies}


\frame{\frametitle{Toward sheaves}

We want to work with sheaves instead of presheaves.

\bloc{External approach}{
Redo [Coquand, Höfer, Sattler 26] with sheaves...
}

\bloc{Internal approach}{
Forget about higher category theory, let's do it in HoTT! 
}

}


 \frame{\frametitle{Internal topologies}
 
 \bloc{Definition}{
A topology consists of a class of finite sums of affines $\J$ stable under $\Sigma$ and $1$.
 }
 
 \bloc{Definition}{
 A $\J$-cover is a map with fibers in $\J$.
 }
 
 %\bloc{Remark}{
 Given $S$ affine, any $\J$-cover is merely of the form
 
 \[U_1+\cdots+U_n \to S\]
 with each $U_i$ representable.
% }

\bloc{Lemma}{
\begin{itemize}
\item Isomorphisms are $\J$-cover.
\item $\J$-covers are stable under pullback and composition.
\end{itemize}
}
 
 }
 
 
 \frame{\frametitle{Internal sheaves}

\bloc{Definiton}{
A type $X$ is a $\J$-sheaf is for all $S:\J$, the map

\[X\to X^{\propTrunc{S}}\]
is an equivalence.
}

This means being a $\J$-sheaf is a lex modality in HoTT [Rijke, Shulman, Spitters 20].

\bloc{Corollary}{
\begin{itemize}
\item $\J$-sheaves are stable under $\Sigma$ and identity types.
\item $\J$-sheaves are stable under $\Pi_X$ for any type $X$.
\item We have a $\J$-sheafification.
\item The type of $\J$-sheaves is a itself a $\J$-sheaf. 
\end{itemize}
}
 
 }
 
 \frame{\frametitle{Why is this reasonable?}

How does this match up with $\J$-sheaves in the usual sense?

\pause

\bloc{Lemma}{
Let $T\to S$ be a $\J$-cover and $X$ be set that is a $\J$-sheaf. Then the map

\vspace{-0.2cm}
\[X^S\to \lim(X^T\rightrightarrows X^{T\times_ST})\]
is an equivalence.
}

\bloc{Remark}{
The is a version for $n$-types based on iterated joins [Williams 25].
}
 
 }
 

 
 
 \frame{\frametitle{Lex modalities and submodels of HoTT}
 
 Lex modalities correspond to $\infty$-subtopoi.
 
So $\J$-sheaves should form a model of HoTT.
 
 \bloc{Theorem [Quirin 16]}{
Given a lex modality $L$, we have an interpretation $\llbracket\_\rrbracket$ of HoTT into $L$-modal types where
\begin{center}
\begin{tabular}{ccc}
$\llbracket \Type \rrbracket$ & $=$ & $\Type_L$ \\
$\llbracket \Sigma_XY \rrbracket$ & $=$ & $\Sigma_{\llbracket X\rrbracket} \llbracket Y \rrbracket$ \\
$\llbracket \Pi_XY \rrbracket$ & $=$ & $\Pi_{\llbracket X\rrbracket} \llbracket Y \rrbracket$ \\
$\llbracket x=_Xy \rrbracket$ & $=$ & $[x]=_{\llbracket X\rrbracket} [y]$\\
$\llbracket \propTrunc{X} \rrbracket$ & $=$ & $L\propTrunc{\llbracket X\rrbracket}$\\
\end{tabular}
\end{center}
 }
 
So we have a $\J$-sheaf model of HoTT. 
 
 }
 
 
 \frame{\frametitle{The sheaf axioms: generic object}
 
 What do the $\J$-sheaf model preserve from the presheaf axioms?
 
 We want that $\G$ is a $\J$-sheaf, giving a generic $\mathbb{T}$-algebra in the $\J$-sheaf model.
 
 \bloc{Definition}{
 A topology $\J$ is subcanonical if $\G$ is a $\J$-sheaf.
 }
 
 \bloc{Remark}{
 By duality, this implies any representable is a $\J$-sheaf.
 }
 
 From now on we assume $\J$ subcanonical.
 
 \bloc{Lemma (Generic object in sheaves)}{
 In the sheaf model, we have that $\propTrunc{S}$ for all $S:\J$.
 }
 
 }
 
 
 \frame{\frametitle{The sheaf axiom: duality}

 \bloc{Lemma (Duality for sheaves)}{
Duality holds in the $\J$-sheaf model.
 }
 
 }
 
 
 \frame{\frametitle{The sheaf axioms: local choice}
 
 \bloc{Definition}{
 A type $X$ has $\J$-local choice if given a surjection $Y\twoheadrightarrow X$, there exists a type $U$ and the dotted arrows in
 
 \[\xymatrix{
 && Y\ar@{->>}[d]\\
U \ar@{-->}[rru]\ar@{-->}[rr]_{\J\mhyphen\text{cover}} && X 
}\]

 }
 
 \bloc{Lemma (Local choice)}{
 In the sheaf model, we have that representable have $\J$-local choice
 }
 
 }


\frame{\frametitle{Summary for sheaf model}

 \bloc{Theorem}{
 Assume given:
 \begin{itemize} 
 \item $\T$ an algebraic theory 
 \item $\J$ a subcanonical topology in HoTT + presheaf axioms.
\end{itemize}
Then there exists a $\J$-sheaf model of HoTT where:
 \begin{itemize}
 \item There is a $\T$-algebra $\G$ such that for all $S:\J$ we have $\propTrunc{S}$.
 \item For all $A:\G\mhyphen\Alg_{f.p.}$, the map $A\to \G^{\Spec(A)}$ is an equivalence.
 \item For all $A:\G\mhyphen\Alg_{f.p.}$, the set $\Spec(A)$ has $\J$-local choice.
 %\item For all $A:\G\mhyphen\Alg_{f.p.}$, the map $\N\to\N^{\Spec(A)}$ is an equivalence.
 \end{itemize}
 }

}

 
\section{Sheaf models for synthetic algebraic geometry and variants}


\frame{\frametitle{The presheaf model}

Ring and algebras over them are assumed commutative.

\bloc{Proposition}{
We have a model of HoTT where:
\begin{itemize}
\item There is a ring $R$.
\item For all $A:R\mhyphen\Alg_{f.p.}$ the map $A\to R^{\Spec(A)}$ is an equivalence.
 \item For all $A:R\mhyphen\Alg_{f.p.}$ the set $\Spec(A)$ has choice.
\end{itemize}
}

}

\frame{\frametitle{Zariski topology}

\bloc{Definition}{
The Zariski topology consists of types of the form

\[r_1\ \text{inv}+\cdots+r_n\ \text{inv} \]
where $(r_1,\cdots,r_n)=1$ in $R$.
}

%\bloc{Definition}{
%An open proposition is a proposition of the form 

%\[r_1\ \text{inv}\lor\cdots\lor r_n\ \text{inv} \]
%where $$ in $r_1,\hdots,r_n:R$.
%}

\bloc{Remark}{
We have $\propTrunc{S}$ for all $S:\text{Zar}$ if and only if $R$ is local.
}

}


\frame{\frametitle{The Zariski sheaf model}

\bloc{Proposition}{
We have a model of HoTT where:
\begin{itemize}
\item There is a local ring $R$.
\item For all $A:R\mhyphen\Alg_{f.p.}$ the map $A\to R^{\Spec(A)}$ is an equivalence.
 \item For all $A:R\mhyphen\Alg_{f.p.}$ the set $\Spec(A)$ has Zariski-local choice.
\end{itemize}
}

}


\frame{\frametitle{An interlude: sheafifying in a sheaf model}

\bloc{Remark}{
We considered topologies internal to presheaves.

The same story works for topologies internal to sheaves.
}

\bloc{Remark}{
In the Zariski sheaf model, finite sums of affines are affines.

So we can replace finite sums of affines by affines in the definition of topology.
}

Now we work in the Zariski sheaf model.

}


\frame{\frametitle{\'Etale topology}

\bloc{Definition}{
A type $X$ is formally \'etale if for all $\epsilon:R$ nilpotent, the map

\[X\to X^{\epsilon=0}\]
is an equivalence.
}

\bloc{Definition}{
The \'etale topology consists of formally \'etale non-empty affines.
}

\bloc{Proposition}{
A type $X$ is an \'etale sheaf if and only if for all $P$ monic unramifiable, the map 

\[X\to X^{\Spec(R[X]/P)}\]
is an equivalence.
}

}


\frame{\frametitle{The \'etale sheaf model}

\bloc{Proposition}{
We have a model of HoTT where:
\begin{itemize}
\item There is a local ring $R$ such that any monic unramifiable polynomial has a root.
\item For all $A:R\mhyphen\Alg_{f.p.}$ the map $A\to R^{\Spec(A)}$ is an equivalence.
 \item For all $A:R\mhyphen\Alg_{f.p.}$ the set $\Spec(A)$ has \'etale-local choice.
\end{itemize}
}

}


\frame{\frametitle{The fppf topology}

\bloc{Definition}{
An $A:R\mhyphen\Alg_f.p.$ is flat if the functor $A\otimes_R\_$ is exact.
}

\bloc{Definition}{
The fppf topology consists of flat non-empty affines.
}

\bloc{Remark}{
Given $P$ monic non-constant, $R[X]/P$ is flat non-empty, 

so in fppf sheaves monic non-constant have roots .
}

\bloc{Remark}{
Formally \'etale affines are flat, so fppf sheaves are \'etale sheaves.
}

}


\frame{\frametitle{The fppf sheaf model}

\bloc{Proposition}{
We have a model of HoTT where:
\begin{itemize}
\item There is a local ring $R$ such that any monic non-constant polynomial has a root.
\item For all $A:R\mhyphen\Alg_{f.p.}$ the map $A\to R^{\Spec(A)}$ is an equivalence.
 \item For all $A:R\mhyphen\Alg_{f.p.}$ the set $\Spec(A)$ has flat-local choice.
\end{itemize}
}

}
