 \section{Introduction to synthetic geometry}
 
 
 \frame{\frametitle{Synthetic geometry}
 
Say we want to work with some geometric objects,

e.g.\ differential manifolds, sequential spaces, schemes...
 
 \bloc{Core idea}{
 \begin{enumerate}
 \item Embeds objects of interests in a topos.
 \item Interpret MLTT + UIP in this topos.
\item Look for new axioms \red{(and modalities)} validated by this interpretation.
\item Work in MLTT + UIP + these axioms \red{(+ modalities)}.
\end{enumerate}
 }
 
 }
 
 
  \frame{\frametitle{What should we expect?}
 
 The objects of interest are types satisfying some proposition.
 
 \bloc{Slogan}{
 \begin{itemize}
 \item All types come with a topology/differential structure/etc.
 \item All maps are continuous/differentiable/etc.
 \end{itemize}
 }

This means we have to abandon LEM and AC.

 \frame{\frametitle{Synthetic homotopy theory}
 
 Say we want to work with homotopy types / $\infty$-groupoids.
 
\bloc{Core idea}{
\begin{enumerate}
\item MLTT can be interpreted in $\infty$-groupoids.
\item This validates HoTT (= univalence + HITs).
\item Work in HoTT.
\end{enumerate}
}
 
 }
 
 
 \frame{\frametitle{We can do both!}
 
 \bloc{Core idea}{
 \begin{enumerate}
 \item Embeds objects of interest in an $\infty$-topos.
 \item Interpret HoTT in this topos.
\item Look for new axioms \red{(and modalities)} validated by this interpretation.
\item Work in HoTT + these axioms \red{(+ modalities)}.
\end{enumerate}
 }

 \bloc{Why do this?}{
 \begin{itemize}
 \item Objects of interests have $\infty$-groupoid-like structure.
 \item We want to use some homotopy theory (e.g.\ cohomology).
 \item For the convenience of univalence and HITs.
 \end{itemize}
 }
 
 }

 \section{Presheaf models}
 
 
 \frame{\frametitle{Goal}
 
This section present results from [Coquand, Höfer, Sattler 26].
 
 Assume given an algebraic theory $\T$, modelling our building blocks. 
 
For SAG, $\T$ is the theory of rings.
 
 \bloc{Goal}{
 Work synthetically in the classifying $\infty$-topos for $\T$.
 }
 
 This is the topos of presheaf over $\T\mhyphen\Alg_{f.p.}$.
 
 \bloc{Idea}{
 Representables are dual to the algebras, and therefore model the basic building blocks.
 }
 
 }
 
 
 \frame{\frametitle{The generic object}
 
 We have a presheaf sending any $A:\T\mhyphen\Alg_{f.p.}$ to its underlying set.
 
 It is a $\T$-algebra in $\PSh(\T\mhyphen\Alg_{f.p.})$.
 
 \bloc{Axiom 1 (Generic object)}{
 There is a set $\G$ with a $\T$-algebra structure.
 }
 
 }
 
 
 \frame{\frametitle{The spectrum construction}
 
 \bloc{Definition}{
 A $\G$-algebra is a $\T$-algebra $A$ with a morphism from $\G$ to $A$.
 }
 
 \bloc{Definition}{
 Given $A:\G\mhyphen\Alg_{f.p.}$, we define
 
 \[\Spec(A) = \Hom_{\G\mhyphen\Alg}(A,\G).\]
 }
 
 \bloc{Example (SAG)}{
% We have a ring $R$. Then
 
% \[\Spec(R[X]) = \Hom_R(R[X],R) = R.\]
 Given $P:R[X]$, then
 
 \[\Spec(R[X]/P) = \Hom_{R\mhyphen Alg}(R[X]/P,R) = \{x:R\ |\ P(x)=0\}.\]
 }
 
 }
 
 
 \frame{\frametitle{Duality axiom and representable}
 
 \bloc{Axiom 2 (Duality)}{
 Assume given $A:\G\mhyphen\Alg_{f.p.}$, then the map
 
 \[A \to R^{\Spec(A)}\]
 is an equivalence.
 }
 
 \bloc{Corollary}{
 The map
 
 \[\Spec : \G\mhyphen\Alg_{f.p.}\to \Set\]
 is an an embedding.
 }
 
The sets in the image of $\Spec$ are called the representables. 
 
In SAG they are called affine schemes.
 
 }
 
 
 \frame{\frametitle{Duality: easy consequences}
 
  \bloc{Corollary}{
 Given $A,B:\G\mhyphen Alg_{f.p.}$, we have that
 
 \[(\Spec(B)\to \Spec(A)) = \Hom_{\G\mhyphen\Alg}(A,B).\]
 }
 
 \bloc{Example (SAG)}{
 \[(R \to R) = \Hom_{R\mhyphen\Alg}(R[X],R[X]) = R[X]\]
 }
 
 }
 
 
 \frame{\frametitle{Choice for representable}
 
 We know that representables in a presheaf category are projective.
 
 \bloc{Axiom 3 (Choice for representable)}{
 Given $A:\G\mhyphen\Alg_{f.p.}$, the type $\Spec(A)$ has choice.
 }
 
 }

  \note{PERHAPS NOT NECESSARY}
 \frame{\frametitle{$\N$ is constant}
 
 $\N$ is interpreted in $\PSh(\T\mhyphen\Alg_{f.p.})$ as the constant presheaf.
 
 \bloc{Axiom 4 ($\N$ constant)}{
 For any $A:\G\mhyphen Alg_{f.p.}$, the map
 
 \[\N\to \N^{\Spec(A)}\]
 is an equivalence.
 }
 
 \note{IS THIS REALLY TRUE (TODO)}
 So family of algebraic objects over $\Spec(A)$ are well-behaved.
 
 \bloc{Proposition}{
 Given $B:\G\mhyphen\Alg_{f.p.}$, the map
 
 \[(B\mhyphen\Alg_{f.p.}) \to (\Spec(B)\to \G\mhyphen\Alg_{f.p.})\]
is an equivalence.
 }
 
 }
 
 
 \frame{\frametitle{The presheaf axioms}
 
 \bloc{Theorem [Coquand, Höfer, Sattler 26]}{
 Given $\T$ an algebraic theory, there exists a model of HoTT plus:
 \begin{itemize}
 \item There is a $\T$-algebra $\G$.
 \item For all $A:\G\mhyphen\Alg_{f.p.}$, the map $A\to \G^{\Spec(A)}$ is an equivalence.
 \item For all $A:\G\mhyphen\Alg_{f.p.}$, the set $\Spec(A)$ has choice.
 \item For all $A:\G\mhyphen\Alg_{f.p.}$, the map $\N\to\N^{\Spec(A)}$ is an equivalence.
 \end{itemize}
 }
 
 }
